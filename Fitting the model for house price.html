# waterfront properties, Ridge Regression

alpha = np.linspace(0,100,1000)
trainig_accuracy = []
test_accuracy = []

for alphavalue in alpha:
    ridge = Ridge(alpha=alphavalue).fit(X_train, y_train)
    trainig_accuracy.append(ridge.score(X_train, y_train).round(2))
    test_accuracy.append(ridge.score(X_test, y_test).round(2))

plt.plot(alpha, trainig_accuracy)
plt.plot(alpha, test_accuracy)
plt.title("model accuracy for waterfront properties\n(Ridge Regression)")
plt.xlabel("alpha")
plt.ylabel("accuracy")
plt.legend(['training', "test"])
plt.savefig("plot3")
plt.show()
print(max(trainig_accuracy))
print(max(test_accuracy))
